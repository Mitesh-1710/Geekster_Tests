Q2. Write a program to check prime number or not.

const number = parseInt(prompt("Enter a number > 1"));

if(number <= 1)
{
   alert("Invalid Input");
}
else
{
	for(let i = 2; i <= number / 2 ; i++)
	{
		if( number % i == 0)
		{
		   console.log(number + " is not a prime number");
		   break;
		}
		else
		{
		   console.log(number + " is a prime number");
		   break;
		}
	}
}
